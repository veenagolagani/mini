Strict Mode Demo

Open the browser console (F12) to see the results!

Non-Strict Mode Example

function nonStrictExample() {
  x = 10; // Creates global variable (bad practice)
  console.log("Non-strict: x =", x);
}
nonStrictExample();

Strict Mode Example

function strictExample() {
  "use strict";
  try {
    x = 10; // Throws ReferenceError
    console.log("Strict: x =", x);
  } catch (e) {
    console.log("Strict mode error:", e.message);
  }
}
strictExample();

Strict Mode Function Scope

function mixedExample() {
  y = 5; // Non-strict, creates global
  console.log("Non-strict: y =", y);
  function innerStrict() {
    "use strict";
    try {
      z = 15; // Throws ReferenceError
      console.log("Strict: z =", z);
    } catch (e) {
      console.log("Strict mode error in function:", e.message);
    }
  }
  innerStrict();
}
mixedExample();
